N = 20

bendingz_modes = cantilever_bending_modes(beam, N)
bendingx_modes = cantilever_bending_modes(beam, N, vertical='x')
axial_modes = cantilever_axial_modes(beam, N)
twisting_modes = cantilever_twisting_modes(beam, N)

all_modes = map(lambda x: (x[0], "bending_z"), bendingz_modes)
all_modes.extend(map(lambda x: (x[0], "bending_x"), bendingx_modes))
all_modes.extend(map(lambda x: (x[0], "axial"), axial_modes))
all_modes.extend(map(lambda x: (x[0], "twisting"), twisting_modes))

# order by ascending frequency
all_modes.sort(key=lambda x: x[0])

for i, (mode, kind) in enumerate(all_modes[:N]):
    print "Mode %i, f = %.2f Hz (%s)" % (i + 1, mode, kind)
